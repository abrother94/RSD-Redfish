)

    $<TARGET_OBJECTS:chassis-status-impl>

    ${LOGGER_LIBRARIES}
    ${PCA95XX_LIBRARIES}

    jsonrpccpp-common

    ${MICROHTTPD_LIBRARIES}
    ${CURL_LIBRARIES}
    ${CY_USB_SERIAL_LIBRARIES}
    ${NET_LIBRARIES}
    ${SAFESTRING_LIBRARIES}
    ${ACC_NET_HELPER_LIBRARIES}
    ${ACC_ONLP_HELPER_LIBRARIES}
